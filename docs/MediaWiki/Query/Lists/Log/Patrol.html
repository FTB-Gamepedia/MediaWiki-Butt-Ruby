def get_patrol_log(user = nil, title = nil, start = nil, stop = nil, limit = @query_limit_default)
  response = get_log('patrol/patrol', user, title, start, stop, limit)

  ret = []
  response['query']['logevents'].each do |log|
    ret << loghash_patrol(log)
  end

  ret
end
